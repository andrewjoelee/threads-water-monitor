function renderUserDetailPage(userId) { document.getElementById('user-view').classList.remove('hidden'); // 【功能一】動態計算粉絲數前三名，特別標註勳章 // 1. 先複製一份 rawData，排除沒有粉絲數的資料，並從大到小排序 const sortedData = [...rawData] .filter(u => u.followers !== undefined && u.followers !== null && u.followers !== "") .sort((a, b) => parseInt(b.followers || 0) - parseInt(a.followers || 0)); // 2. 找出目前點進來的這台飲水機 const item = rawData.find(u => u.id === userId); if (!item) { document.getElementById('det-name').innerText = "找不到該飲水機資料"; return; } // 3. 在排序後的陣列裡，找出這台飲水機是第幾名 (索引 + 1) const rank = sortedData.findIndex(u => u.id === userId) + 1; // 【功能二】處理稱號：刪除後面的「學校」，無稱號就空白，前三名加勳章 let titleText = item.title || ""; // 抓取稱號，如果沒有就空白 // 如果稱號結尾是「學校」或「社員」，就把這兩個字刪掉，讓字面更乾淨 if (titleText.endsWith("學校")) { titleText = titleText.slice(0, -2); } else if (titleText.endsWith("社員")) { titleText = titleText.slice(0, -2); } // 如果這台飲水機名列前三名，且稱號不為空，就自動補上獎牌標註 if (titleText !== "") { if (rank === 1) { titleText = "🥇 冠軍 " + titleText; } else if (rank === 2) { titleText = "🥈 亞軍 " + titleText; } else if (rank === 3) { titleText = "🥉 季軍 " + titleText; } } // 渲染稱號到網頁上 const titleEl = document.getElementById('det-title-link'); // 或者是你的稱號標籤 ID if (titleEl) { titleEl.innerText = titleText; // 【無稱號就空白】如果稱號完全是空的，直接把外層的黃色/藍色背景框隱藏起來 if (titleText === "") { titleEl.parentElement.style.display = "none"; } else { titleEl.parentElement.style.display = "inline-block"; } } // 填入飲水機名稱與正確的小寫 imageUrl 圖片 document.getElementById('det-name').innerText = item.name || "未命名飲水機"; document.getElementById('det-img').src = item.imageUrl || "https://via.placeholder.com/400x400?text=No+Image"; // 【功能三】將單頁的 "ID" 欄位標題改為 "Threads" // 直接用 JS 去抓取 ID 連結前面的文字標籤，強制改成 Threads const idLink = document.getElementById('det-id-link'); if (idLink) { idLink.innerText = "@" + (item.id || "-"); // 順便修正前面的 標題字樣 const idLabel = idLink.parentElement.previousElementSibling; if (idLabel && (idLabel.innerText === "ID" || idLabel.innerText.includes("ID"))) { idLabel.innerText = "Threads"; } } // 填入地址與地圖連結 const locLink = document.getElementById('det-location-link'); if (locLink) { locLink.innerText = item.location || "未提供詳細地址"; if (item.mapUrl) { locLink.href = item.mapUrl; locLink.style.pointerEvents = "auto"; locLink.style.textDecoration = "underline"; } else { locLink.removeAttribute('href'); locLink.style.pointerEvents = "none"; locLink.style.textDecoration = "none"; } } // 填入學校名稱與連結 const schoolLink = document.getElementById('det-school-link'); if (schoolLink) { schoolLink.innerText = item.school || "-"; if (item.schoolUrl) { schoolLink.href = item.schoolUrl; schoolLink.style.pointerEvents = "auto"; schoolLink.style.textDecoration = "underline"; schoolLink.style.color = "#0066cc"; } else { schoolLink.removeAttribute('href'); schoolLink.style.pointerEvents = "none"; schoolLink.style.textDecoration = "none"; schoolLink.style.color = "#1a1a1a"; } } // 填入品牌名稱與連結 const brandLink = document.getElementById('det-brand-link'); if (brandLink) { brandLink.innerText = item.brand || "未標記"; if (item.brandUrl) { brandLink.href = item.brandUrl; } else { brandLink.removeAttribute('href'); } } // 填入最新粉絲數量與更新時間 const followersEl = document.getElementById('det-followers'); if (followersEl) { // 格式化數字加上千分位（例如 2700 改成 2,700） const num = parseInt(item.followers || 0); followersEl.innerText = num.toLocaleString(); } const updateTimeEl = document.getElementById('det-update-time'); if (updateTimeEl) { updateTimeEl.innerText = item.updateTime ? "數據更新時間：" + item.updateTime : ""; } }
